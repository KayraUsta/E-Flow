    INST_DATA_NUMERIC AAD ON AAD.CIID = CRS.CIID AND AAD.DID = 10453
LEFT OUTER JOIN 											  
    INST_DATA_DATE AAF ON AAF.CIID = CRS.CIID AND AAF.DID = 10449

SELECT
    CRS.CIID AS 'Süreç No',
    AAA.VALUE AS 'Fatura Ünvanı',
    AAS.VALUE AS 'Fatura No',
    AAD.VALUE AS 'Fatura Tutarı',
    AAF.VALUE AS 'Fatura Tarihi',
    -- AAG.VALUE AS 'Personel Seç',
    AAH.VALUE AS 'Döküman 1',
    AAJ.VALUE AS 'Döküman 2'
FROM 
    INST_COURSE CRS
LEFT OUTER JOIN 
    INST_DATA_STRING AAA ON AAA.CIID = CRS.CIID AND AAA.DID = 10451
LEFT OUTER JOIN 
    INST_DATA_STRING AAS ON AAS.CIID = CRS.CIID AND AAS.DID = 10450
LEFT OUTER JOIN 											  
    INST_DATA_NUMERIC AAD ON AAD.CIID = CRS.CIID AND AAD.DID = 10453
LEFT OUTER JOIN 											  
    INST_DATA_DATE AAF ON AAF.CIID = CRS.CIID AND AAF.DID = 10449
-- LEFT OUTER JOIN 
    -- INST_DATA_STRING AAG ON AAG.CIID = CRS.CIID AND AAG.DID = 10051
LEFT OUTER JOIN 											  
    INST_DATA_ATTACHMENT AAH ON AAH.CIID = CRS.CIID AND AAH.DID = 10454
LEFT OUTER JOIN 											 
    INST_DATA_ATTACHMENT AAJ ON AAJ.CIID = CRS.CIID AND AAJ.DID = 10052
WHERE 
    AAA.VALUE IS NOT NULL OR
    AAS.VALUE IS NOT NULL OR
    AAD.VALUE IS NOT NULL OR
    AAF.VALUE IS NOT NULL OR
    -- AAG.VALUE IS NOT NULL OR
    AAH.VALUE IS NOT NULL OR
    AAJ.VALUE IS NOT NULL;
